# Install
pip install trustlayer

# Start the server + UI
trustlayer server

# Auto-detects Ollama, checks for API keys
# Opens http://localhost:8000
# Verify any AI output
trustlayer verify "The earth is 4.5 billion years old."

# Ask any connected AI
trustlayer ask "Summarize this PR" --provider ollama --model llama3.2

# Compare across providers
trustlayer compare "Write a regex for email validation"

# Check your spending
trustlayer costs
# Verify via REST API
curl -X POST http://localhost:8000/api/verify \
  -H "Content-Type: application/json" \
  -d '{"content": "Your AI output here"}'

# Response:
{
  "trust_score": 87,
  "trust_label": "high",
  "summary": "This response is 87% trusted.",
  "issues": []
}
